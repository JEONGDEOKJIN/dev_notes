
## 메타 데이터를 설정하는 이유가 뭘까? 
```
SEO	: Google, Naver 검색
소셜 공유	: 카카오톡, 페이스북, 트위터
브랜딩 : 파비콘, 탭 제목 
```

## 예시 
```
export const metadata: Metadata = {
    // 브라우저 탭 제목 + 검색 결과 제목
    title: '페이지 제목 | 브랜드명',

    // 파비콘 (브라우저 탭 아이콘)
    icons: {
        icon: `${process.env.NEXT_PUBLIC_KIWEB_STATIC_URL}assets/event/favicon.png`,
    },

    // 검색 결과에 표시되는 설명문 (150자 내외 권장)
    description: '페이지에 대한 간단한 설명. 검색 결과에 노출됩니다.',

    // 검색 엔진 키워드 (SEO 효과는 낮지만 참고용)
    keywords: [
        '키워드1',
        '키워드2',
        '키워드3',
        '관련 서비스',
        '브랜드명',
    ],

    // Open Graph: 카카오톡, 페이스북, 슬랙 등 소셜 미디어 공유용
    openGraph: {
        // 공유 시 제목
        title: '소셜 공유용 제목',

        // 공유 시 설명
        description: '소셜 미디어에서 링크 공유 시 보이는 설명입니다.',

        // 페이지 URL
        url: 'https://example.com/page',

        // 콘텐츠 타입 (website, article, product 등)
        type: 'website',

        // 사이트 이름
        siteName: '사이트명',

        // 언어/지역 설정
        locale: 'ko_KR',

        // 썸네일 이미지 (1200x630px 권장)
        images: [
            {
                url: `${process.env.NEXT_PUBLIC_KIWEB_STATIC_URL}assets/event/thumbnail.png`,
            },
        ],
    },

    // Twitter(X) 공유용
    twitter: {
        // 카드 타입 (summary_large_image: 큰 이미지, summary: 작은 이미지)
        card: 'summary_large_image',

        // 트위터 공유 시 제목
        title: '트위터 공유용 제목',

        // 트위터 공유 시 설명
        description: '트위터에서 링크 공유 시 보이는 설명입니다.',

        // 썸네일 이미지
        images: [
            {
                url: `${process.env.NEXT_PUBLIC_KIWEB_STATIC_URL}assets/event/thumbnail.png`,
            },
        ],
    },
}
```
